122  virtual inline double f(double y)
123  {
124  double t;
125  if (F_tau == 0) {
126  t = 1;

134  virtual inline double rk4(double y_n, double step)
135  {
136  double h=step/2.0, /* the midpoint */
137  t1, t2, t3, /* temporary storage arrays */
138  k1, k2, k3,k4; /* for Runge-Kutta */
139 
140  if (fabs(y_n) > MAXFLOAT){
141  return 1.0;
142  }
143 
144  t1=y_n+0.5*(k1=step*f(y_n));
145  t2=y_n+0.5*(k2=step*f(t1));
146  t3=y_n+ (k3=step*f(t2));
147  k4= step*f(t3);
148 
149  return y_n+=(k1+2*k2+2*k3+k4)/6.0;
150  }

226  double __fastcall Update(double V, double step) //step is ms, V is mV.
227  {
228  /* DONE -oebt-farzan : Here is where we can loop to increase number of
229  integrations per step. Should add stability. */
230  F_tau = tau(V);
231  F_inf = inf(V);
232 
233  double max_step = GetNet()->GetMaxRK4Timestep();
234  if (step > max_step){
235  int numRK4calls = floor(step / max_step);
236  double leftover = fmod (step , max_step);
237  for (int i = 0; i < numRK4calls; i++) {
238  F_y_n = rk4(F_y_n, max_step);
239  }
240  if (leftover > 0.0) {
241  F_y_n = rk4(F_y_n, leftover);
242  }
243  }
244  else {
245  F_y_n = rk4(F_y_n, step);
246  }
247 
248  return F_y_n;
249  }

252  virtual inline double tau(double V)
253  {
254  // with error checks and constraints
255  double arg, k = fabs(F_k);
256  if ( F_k == 0 ) return F_t_lo;
257  arg = (V - F_V0) / k;
258  if(arg > 10) return F_t_hi;
259  else if (arg < -10) return F_t_lo;
260  else {
261  if(F_w == 1.0) return ( F_t_hi + ( F_t_lo - F_t_hi ) / (1+exp(arg)) );
262  else if (F_w > 0) return ( F_t_hi + ( F_t_lo - F_t_hi ) / pow(1+exp(arg), F_w) );
263  else return 1.0;
264  }
265  }
266 
268  virtual inline double inf(double V)
269  {
270  // with error checks and constraints
271  double arg;
272  if(F_k == 0) return F_infMin;
273  arg=(V-F_V0)/F_k;
274  if(arg > 10) return F_infMin;
275  else if (arg < -10) return 1.0;
276  if (F_w != 1.0 && F_w > 0) arg = pow(1+exp(arg), F_w);
277  else arg = 1 + exp(arg);
278  return ( (1 - F_infMin)/arg + F_infMin);
279  }
